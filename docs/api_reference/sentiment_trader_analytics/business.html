def export_insights_to_markdown( insights: list[BusinessInsight], output_path: pathlib._local.Path) -> None: 
302def export_insights_to_markdown(
303    insights: list[BusinessInsight],
304    output_path: Path,
305) -> None:
306    """Export a list of validated insights to a formatted Markdown document.
307
308    Only insights with ``report_ready: True`` are included. The document
309    contains a header, metadata table, and the five-part body for each
310    insight.
311
312    Args:
313        insights: List of :class:`BusinessInsight` objects to export.
314        output_path: Filesystem path for the output ``.md`` file.
315
316    Raises:
317        ValueError: If no report-ready insights are provided.
318    """
319    report_ready = [i for i in insights if i.report_ready]
320    if not report_ready:
321        raise ValueError(
322            "Cannot export: no report-ready insights. "
323            "Run validate_insight on each insight first."
324        )
325
326    output_path.parent.mkdir(parents=True, exist_ok=True)
327
328    header = [
329        "# Business Insights Draft",
330        "",
331        f"**Generated:** {__import__('datetime').datetime.now().strftime('%Y-%m-%d %H:%M:%S')}",
332        f"**Report-Ready Insights:** {len(report_ready)} / {len(insights)}",
333        "",
334        "---",
335        "",
336    ]
337
338    toc_parts = ["## Table of Contents", ""]
339    body_sections = []
340
341    # Group insights by act
342    acts: dict[str, list[BusinessInsight]] = {
343        "Act I — The Setup": [],
344        "Act II — The Conflict": [],
345        "Act III — The Resolution": [],
346        "Additional Findings": [],
347    }
348
349    for ins in report_ready:
350        act = _NARRATIVE_ACTS.get(ins.insight_id, "Additional Findings")
351        if act not in acts:
352            acts["Additional Findings"].append(ins)
353        else:
354            acts[act].append(ins)
355
356    for act_name, act_insights in acts.items():
357        if not act_insights:
358            continue
359
360        toc_parts.append(f"### {act_name}")
361        body_sections.append(f"# {act_name}")
362        body_sections.append("")
363
364        for ins in act_insights:
365            anchor = ins.insight_id.lower()
366            display_title = (
367                ins.title if ins.title else (ins.statistical_evidence.metric or ins.insight_id)
368            )
369            toc_parts.append(f"- [{ins.insight_id}: {display_title}](#{anchor})")
370            body_sections.append(_insight_to_markdown(ins))
371
372        toc_parts.append("")
373
374    toc_parts.append("---")
375    toc_parts.append("")
376
377    all_lines = header + toc_parts + body_sections
378    content = "\n".join(all_lines)
379
380    output_path.write_text(content, encoding="utf-8")
381    logger.info(
382        "Exported %d report-ready insights to %s",
383        len(report_ready),
384        output_path,
385    )

def validate_insight( insight: BusinessInsight) -> bool: 
178def validate_insight(insight: BusinessInsight) -> bool:
179    """Validate that a :class:`BusinessInsight` has all required components.
180
181    Checks:
182        1. All five string components are non-empty.
183        2. The underlying ``effect_size`` in statistical_evidence is not None.
184
185    Sets ``insight.report_ready`` to the result of validation.
186
187    Args:
188        insight: The :class:`BusinessInsight` to validate.
189
190    Returns:
191        ``True`` if the insight is report-ready, ``False`` otherwise.
192    """
193    required_text_fields = [
194        ("observation", insight.observation),
195        ("business_interpretation", insight.business_interpretation),
196        ("practical_recommendation", insight.practical_recommendation),
197        ("limitation", insight.limitation),
198        ("insight_id", insight.insight_id),
199    ]
200
201    for field_name, value in required_text_fields:
202        if not isinstance(value, str) or not value.strip():
203            logger.warning(
204                "Insight %s validation failed: field '%s' is empty",
205                insight.insight_id,
206                field_name,
207            )
208            insight.report_ready = False
209            return False
210
211    if insight.statistical_evidence.effect_size is None:
212        logger.warning(
213            "Insight %s validation failed: effect_size is None",
214            insight.insight_id,
215        )
216        insight.report_ready = False
217        return False
218
219    insight.report_ready = True
220    return True

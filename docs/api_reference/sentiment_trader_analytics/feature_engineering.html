sentiment_trader_analytics.feature_engineering

Feature engineering module for the sentiment trader analytics pipeline.

Provides pure, stateless functions for constructing engineered features across three domains: - Sentiment features (sentiment_features) - Trader performance features (trader_features) - Time-based features (time_features)

All feature functions follow the signature f(df: pd.DataFrame, config: FeatureConfig) -> pd.DataFrame.

17def add_sentiment_lag(df: pd.DataFrame, config: FeatureConfig) -> pd.DataFrame:
18    """Add a 1-day lagged sentiment value column.
19

21    look-ahead bias — the lagged value comes strictly from a prior row.
22
23    Args:
24        df: DataFrame with a ``sentiment_value`` column.
25        config: Feature engineering configuration.
26
27    Returns:
28        DataFrame with ``sentiment_value_lag_1d`` column added.
29    """
30    result = df.copy()
31    result["sentiment_value_lag_1d"] = result["sentiment_value"].shift(config.sentiment_lag_days)
32    return result

def add_sentiment_regime_encoding( df: pandas.core.frame.DataFrame, config: sentiment_trader_analytics.config.FeatureConfig) -> pandas.core.frame.DataFrame: 
35def add_sentiment_regime_encoding(df: pd.DataFrame, config: FeatureConfig) -> pd.DataFrame:
36    """Ordinal-encode the sentiment classification column.
37
38    Mapping is defined in ``config.sentiment_regime_encoding``:
39        Extreme Fear → 0, Fear → 1, Neutral → 2, Greed → 3, Extreme Greed → 4
40
41    Rows with missing classification produce NaN in the encoding.
42
43    Args:
44        df: DataFrame with a ``sentiment_classification`` column.
45        config: Feature engineering configuration.
46
47    Returns:
48        DataFrame with ``sentiment_regime_encoded`` column added.
49    """
50    result = df.copy()
51    encoded = result["sentiment_classification"].map(config.sentiment_regime_encoding)
52    # Rows with missing sentiment classification (e.g., no sentiment data
53    # for that date) produce NaN. We fill with Neutral (2) as a neutral
54    # default rather than leaving a nullable int column that pandas cannot
55    # represent as int8.
56    if encoded.isna().any():
57        encoded = encoded.fillna(2.0)
58    result["sentiment_regime_encoded"] = encoded.astype(np.int8)
59    return result

Ordinal-encode the sentiment classification column.

Mapping is defined in config.sentiment_regime_encoding: Extreme Fear → 0, Fear → 1, Neutral → 2, Greed → 3, Extreme Greed → 4

Rows with missing classification produce NaN in the encoding.

Arguments:
  • df: DataFrame with a sentiment_classification column.
  • config: Feature engineering configuration.
Returns:

DataFrame with sentiment_regime_encoded column added.

62def add_sentiment_rolling_mean(df: pd.DataFrame, config: FeatureConfig) -> pd.DataFrame:
63    """Add a 7-day rolling mean of the sentiment value.
64
65    Uses ``closed="left"`` to exclude the current row from the window,
66    preventing any look-ahead bias.
67
68    Args:
69        df: DataFrame with a ``sentiment_value`` column.
70        config: Feature engineering configuration.
71
72    Returns:
73        DataFrame with ``sentiment_value_rolling_7d`` column added.
74    """
75    result = df.copy()
76    result["sentiment_value_rolling_7d"] = (
77        result["sentiment_value"]
78        .rolling(window=config.sentiment_rolling_window, min_periods=1, closed="left")
79        .mean()
80    )
81    return result

def add_trader_win_rate( df: pandas.core.frame.DataFrame, config: sentiment_trader_analytics.config.FeatureConfig) -> pandas.core.frame.DataFrame: 
45def add_trader_win_rate(df: pd.DataFrame, config: FeatureConfig) -> pd.DataFrame:
46    """Add rolling 7-day win rate per account.
47
48    Win rate is the fraction of closed trades (``Direction == "Close"``)
49    with positive ``Closed PnL`` within the rolling window.
50    Uses ``closed="left"`` to exclude current row.
51
52    Args:
53        df: DataFrame with ``Account``, ``Direction``, and ``Closed PnL`` columns.
54        config: Feature engineering configuration.
55
56    Returns:
57        DataFrame with ``trader_win_rate_7d`` column added.
58    """
59    window = config.trader_rolling_windows["win_rate"]
60    result = df.copy()
61    is_close = result["Direction"].astype(str).str.contains("Close", case=False, na=False)
62    result["_is_win"] = (is_close & (result["Closed PnL"] > 0)).astype(float)
63    result["_is_trade"] = (is_close & (result["Closed PnL"].notna())).astype(float)
64    result["trader_win_rate_7d"] = (
65        result.groupby("Account")["_is_win"].transform(
66            lambda s: s.rolling(window=window, min_periods=1, closed="left").sum()
67        )
68    ) / (
69        result.groupby("Account")["_is_trade"].transform(
70            lambda s: s.rolling(window=window, min_periods=1, closed="left").sum()
71        )
72    )
73    result = result.drop(columns=["_is_win", "_is_trade"])
74    return result

Add rolling 7-day win rate per account.

Win rate is the fraction of closed trades (Direction == "Close") with positive Closed PnL within the rolling window. Uses closed="left" to exclude current row.

Arguments:
  • df: DataFrame with Account, Direction, and Closed PnL columns.
  • config: Feature engineering configuration.
Returns:

DataFrame with trader_win_rate_7d column added.

def flag_cold_start_rows( df: pandas.core.frame.DataFrame, config: sentiment_trader_analytics.config.FeatureConfig) -> pandas.core.frame.DataFrame: 
178def flag_cold_start_rows(df: pd.DataFrame, config: FeatureConfig) -> pd.DataFrame:
179    """Flag rows where rolling trader features have not yet warmed up.
180
181    A row is flagged as cold-start if ANY of the rolling trader feature
182    columns is NaN for that row. Cold-start rows are retained for
183    completeness but should be excluded from downstream statistical
184    analyses that require fully-warmed windows.
185
186    Args:
187        df: DataFrame with rolling trader feature columns.
188        config: Feature engineering configuration (unused, kept for signature).
189
190    Returns:
191        DataFrame with ``feature_cold_start`` boolean column added.
192    """
193    _ = config
194    result = df.copy()
195    rolling_cols = [col for col in result.columns if col.startswith("trader_")]
196    result["feature_cold_start"] = result[rolling_cols].isna().any(axis=1)
197    return result

Flag rows where rolling trader features have not yet warmed up.

A row is flagged as cold-start if ANY of the rolling trader feature columns is NaN for that row. Cold-start rows are retained for completeness but should be excluded from downstream statistical analyses that require fully-warmed windows.

Arguments:
  • df: DataFrame with rolling trader feature columns.
  • config: Feature engineering configuration (unused, kept for signature).
Returns:

DataFrame with feature_cold_start boolean column added.

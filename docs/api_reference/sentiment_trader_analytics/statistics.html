def apply_multiple_testing_correction( results: list[StatTestResult], method: str = 'bonferroni', alpha: float = 0.05) -> list[StatTestResult]: 
630def apply_multiple_testing_correction(
631    results: list[StatTestResult],
632    method: str = "bonferroni",
633    alpha: float = 0.05,
634) -> list[StatTestResult]:
635    """Apply multiple testing correction to a list of test results.
636
637    Args:
638        results: List of StatTestResult objects to correct.
639        method: Correction method ("bonferroni" or "fdr_bh").
640        alpha: Family-wise significance level (default 0.05).
641
642    Returns:
643        The input list with correction_applied and corrected_threshold set.
644    """
645    n_tests = len(results)
646    if n_tests == 0:
647        return results
648
649    if method == "bonferroni":
650        corrected_threshold = alpha / n_tests
651        for result in results:
652            result.correction_applied = "bonferroni"
653            result.corrected_threshold = corrected_threshold
654    elif method == "fdr_bh":
655        p_vals = [r.p_value for r in results]
656        _rejected, _corrected_pvals, _alphac_sidak, _alphac_bonf = multipletests(
657            p_vals, alpha=alpha, method="fdr_bh"
658        )
659        corrected_threshold = float(np.min([alpha * (i + 1) / n_tests for i in range(n_tests)]))
660        for result in results:
661            result.correction_applied = "fdr_bh"
662            result.corrected_threshold = corrected_threshold
663    else:
664        raise ValueError(f"Unknown correction method: {method}")
665
666    return results

 85    """Test whether a series follows a normal distribution.
 86
 87    Uses Shapiro-Wilk for n <= 50 and D'Agostino-Pearson for n > 50.
 88
 89    Args:
 90        series: The data series to test.
 91        alpha: Significance level for the normality test (default 0.05).
 92
 93    Returns:
 94        A NormalityResult with statistic, p-value, method, and decision.
 95
 96    Raises:
 97        ValueError: If the series has fewer than 3 non-NA observations.
 98    """
 99    data = series.dropna()
100    n = len(data)
101    if n < 3:
102        raise ValueError(
103            f"Normality test requires at least 3 observations; got {n}"
104        )
105
106    if n <= 50:
107        stat_val, p_val = shapiro(data)
108        method = "shapiro"
109    else:
110        stat_val, p_val = normaltest(data)
111        method = "dagostino_pearson"
112
113    return NormalityResult(
114        statistic=float(stat_val),
115        p_value=float(p_val),
116        method=method,
117        normality_rejected=bool(p_val < alpha),
118    )

486def compare_multiple_groups(
487    groups: dict[str, pd.Series],
488    alpha: float = 0.05,
489    min_sample_size: int = 30,
490) -> StatTestResult:
491    """Compare multiple independent groups with appropriate test selection.
492
493    Protocol:
494        1. Normality pre-test on all groups.
495        2. If all groups normal -> ANOVA + Tukey HSD.
496           Otherwise -> Kruskal-Wallis + Dunn's test with Holm-Bonferroni.
497        3. Effect size: eta-squared (ANOVA) or epsilon-squared (Kruskal-Wallis).
498
499    Args:
500        groups: Map of group name to data series.
501        alpha: Significance level (default 0.05).
502        min_sample_size: Minimum sample size per group (default 30).
503
504    Returns:
505        A StatTestResult with the test outcome.
506    """
507    cleaned: dict[str, pd.Series] = {}
508    for name, series in groups.items():
509        cleaned[name] = series.dropna()
510
511    sample_sizes = {name: len(s) for name, s in cleaned.items()}
512    underpowered = any(n < min_sample_size for n in sample_sizes.values())
513
514    any_nonnormal = False
515    for _name, series in cleaned.items():
516        nr = check_normality(series, alpha)
517        if nr.normality_rejected:
518            any_nonnormal = True
519            break
520
521    group_series_list = list(cleaned.values())
522    group_names = list(cleaned.keys())
523
524    if not any_nonnormal:
525        stat_val, p_val = f_oneway(*group_series_list)
526        test_name = "anova_one_way"
527        effect_size, ci = _eta_squared(cleaned)
528        effect_size_measure = "eta_squared"
529        posthoc = _tukey_hsd_posthoc(cleaned, alpha)
530    else:
531        stat_val, p_val = kruskal(*group_series_list)
532        test_name = "kruskal_wallis"
533        effect_size, ci = _epsilon_squared(cleaned)
534        effect_size_measure = "epsilon_squared"
535        posthoc = _dunn_posthoc(cleaned, alpha)
536
537    logger.debug("Post-hoc results for %s: %s", test_name, posthoc)
538
539    return StatTestResult(
540        test_name=test_name,
541        metric="unknown",
542        groups=group_names,
543        statistic=float(stat_val),
544        p_value=float(p_val),
545        effect_size=effect_size,
546        effect_size_measure=effect_size_measure,
547        confidence_interval_95=ci,
548        sample_sizes=sample_sizes,
549        normality_rejected=any_nonnormal,
550        underpowered=underpowered,
551    )

def compare_two_groups( group_a: pandas.core.series.Series, group_b: pandas.core.series.Series, alpha: float = 0.05, min_sample_size: int = 30) -> StatTestResult: 
191def compare_two_groups(
192    group_a: pd.Series,
193    group_b: pd.Series,
194    alpha: float = 0.05,
195    min_sample_size: int = 30,
196) -> StatTestResult:
197    """Compare two independent groups with appropriate test selection.
198
199    Protocol:
200        1. Normality pre-test on both groups.
201        2. Variance homogeneity test (Levene's).
202        3. If both normal and equal variance -> independent t-test.
203           If both normal but unequal variance -> Welch's t-test.
204           Otherwise -> Mann-Whitney U.
205        4. Effect size: Cohen's d (parametric) or rank-biserial r (non-parametric).
206
207    Args:
208        group_a: Data for the first group.
209        group_b: Data for the second group.
210        alpha: Significance level (default 0.05).
211        min_sample_size: Minimum sample size per group (default 30).
212
213    Returns:
214        A StatTestResult with the test outcome.
215    """
216    a_clean = group_a.dropna()
217    b_clean = group_b.dropna()
218
219    normality_a = check_normality(a_clean, alpha)
220    normality_b = check_normality(b_clean, alpha)
221    normality_rejected = normality_a.normality_rejected or normality_b.normality_rejected
222
223    underpowered = len(a_clean) < min_sample_size or len(b_clean) < min_sample_size
224
225    both_normal = not normality_a.normality_rejected and not normality_b.normality_rejected
226
227    if both_normal:
228        var_equal = True
229        if len(a_clean) > 1 and len(b_clean) > 1:
230            _levene_stat, levene_p = levene(a_clean, b_clean)
231            var_equal = levene_p >= alpha
232
233        if var_equal:
234            stat_val, p_val = ttest_ind(a_clean, b_clean, equal_var=True)
235            test_name = "independent_t_test"
236        else:
237            stat_val, p_val = ttest_ind(a_clean, b_clean, equal_var=False)
238            test_name = "welch_t_test"
239
240        effect_size, _ci = _cohens_d(a_clean, b_clean)
241        effect_size_measure = "cohens_d"
242
243        diff_ci = _mean_diff_ci(a_clean, b_clean, alpha)
244    else:
245        stat_val, p_val = mannwhitneyu(a_clean, b_clean, alternative="two-sided")
246        test_name = "mann_whitney_u"
247
248        effect_size, _ci = _rank_biserial_r(a_clean, b_clean)
249        effect_size_measure = "rank_biserial_r"
250
251        diff_ci = _median_diff_ci(a_clean, b_clean)
252
253    sample_sizes = {"group_a": len(a_clean), "group_b": len(b_clean)}
254
255    return StatTestResult(
256        test_name=test_name,
257        metric=group_a.name or "unknown",
258        groups=["group_a", "group_b"],
259        statistic=float(stat_val),
260        p_value=float(p_val),
261        effect_size=effect_size,
262        effect_size_measure=effect_size_measure,
263        confidence_interval_95=diff_ci,
264        sample_sizes=sample_sizes,
265        normality_rejected=normality_rejected,
266        underpowered=underpowered,
267    )
